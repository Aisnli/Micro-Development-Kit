	port = 8888
	[/ser config]

	string ip = cfg["ser config"]["ip"];
	cfg["ser config"]["ip"] = "127.0.0.1";
	cfg["ser config"]["port"] = 8080;//可以直接使用char*、string、int、float等进行赋值
